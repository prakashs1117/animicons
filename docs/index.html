2 Basic Usage — React
import { Bell, Refresh, Check } from '@animicons/react'; export default function App() { return ( <div> <Bell /> <Refresh color="#6366f1" size={32} speed="fast" /> <Check loop={false} onAnimationEnd={() => console.log('done')} /> </div> ); }
2 Basic Usage — React Native
import { Bell, Refresh, Check } from '@animicons/react-native'; import { View } from 'react-native'; export default function App() { return ( <View> <Bell /> <Refresh color="#6366f1" size={32} speed="fast" /> <Check loop={false} onAnimationEnd={() => console.log('done')} /> </View> ); }
3 React Native — babel.config.js
// babel.config.js — add the Reanimated plugin (must be last) module.exports = { presets: ['module:metro-react-native-babel-preset'], // or 'babel-preset-expo' for Expo projects plugins: ['react-native-reanimated/plugin'], };
4 Color & Style Customisation
// All props are optional — every icon has sensible defaults <Bell color="#6366f1" // base color (stroke + fill) strokeColor="#818cf8" // override stroke independently fillColor="transparent" // override fill independently secondaryColor="#6366f133" // accent fill (default = base @ 20% alpha) opacity={0.9} // 0–1 strokeWidth={2} // override line thickness size={48} // width & height in px speed="normal" // 'slow' | 'normal' | 'fast' autoPlay={true} // start animation immediately loop={true} // repeat forever onAnimationEnd={() => {}} // called when loop=false finishes />
